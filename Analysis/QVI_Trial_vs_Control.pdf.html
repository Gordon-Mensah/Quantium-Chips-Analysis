def calculate_similarity(trial_store, control_store, data):
    trial_data = data[data['STORE_NBR'] == trial_store]
    control_data = data[data['STORE_NBR'] == control_store]
    return trial_data['Total_Sales'].corr(control_data['Total_Sales'])

# Example: Compare trial store 77 to other stores
similarity_scores = {
    store: calculate_similarity(77, store, store_metrics)
    for store in store_metrics['STORE_NBR'].unique() if store != 77
}

print(similarity_scores)
